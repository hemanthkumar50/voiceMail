<script>
  document.addEventListener('DOMContentLoaded', function() {
      const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
      const synth = window.speechSynthesis; // Get the speechSynthesis object

      if (SpeechRecognition && synth) {
          let recognition;
          let currentInput = null; // Track the currently focused input

          // Function to initiate speech recognition
          function initRecognition() {
              recognition = new SpeechRecognition();
              recognition.continuous = false; // Set to false for discrete recognition
              recognition.lang = 'en-US'; // Specify the language
              recognition.interimResults = false; // We are only interested in final results

              recognition.onresult = function(event) {
                  // Concatenate results if recognition is triggered multiple times
                  const transcript = Array.from(event.results)
                    .map(result => result[0].transcript)
                    .join('');
                  if (currentInput && transcript) {
                      currentInput.value += transcript; // Append new text to the current input
                  }
              };

              recognition.onend = function() {
                  // Restart recognition for new input
                  if (currentInput) {
                      initRecognition();
                      recognition.start();
                  }
              };

              recognition.onerror = function(event) {
                  console.error('Speech recognition error: ' + event.error);
              };
          }

          // Function to stop speech synthesis
          function stopSpeaking() {
              if (synth.speaking) { // If speech is currently being spoken
                  synth.cancel(); // Cancel the speech
              }
          }

          document.querySelectorAll('input[type="text"], input[type="email"], textarea').forEach(input => {
              input.addEventListener('mouseover', function() {
                  stopSpeaking(); // Ensure to stop any ongoing speech first
                  const msg = new SpeechSynthesisUtterance('You are now focused on the ' + this.placeholder + ' field. Please speak your input after the beep.');
                  synth.speak(msg);
              });

              input.addEventListener('mouseout', function() {
                  stopSpeaking(); // Stop speaking when mouse leaves the input
              });

              input.addEventListener('click', function() {
                  if (currentInput !== this) {
                      currentInput = this; // Update the current input being focused
                      if (recognition) {
                          recognition.abort(); // Abort any existing recognition session
                      }
                      initRecognition(); // Initialize a new recognition instance
                      recognition.start(); // Start recognition
                  }
              });
          });
      } else {
          console.log('Speech recognition or synthesis not supported in this browser.');
      }
  });
</script>
